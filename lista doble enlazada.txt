#lista doblemente enlazada

class Nodo():
  def __init__(self, dato):
    self.dato = dato
    self.siguiente = None
    self.anterior = None

class ListaDobleEnlazada():
  def __init__(self):
    self.primero = None
    self.ultimo = None
    self.tamaño = 0

  def agregar_final(self, dato):
    if self.vacia():
      self.primero = self.ultimo = Nodo(dato)
    else:
      aux = self.ultimo
      self.ultimo = aux.siguiente = Nodo(dato)
      self.ultimo.anterior = aux

    self.tamaño += 1

  def agregar_inicio(self, dato):
    if self.vacia():
      self.primero = self.ultimo = Nodo(dato)
    else:
      aux = Nodo(dato)
      aux.siguiente = self.primero
      self.primero.anterior = aux
      self.primero = aux
    self.tamaño += 1

  
  def eliminar_primero(self):
      if not self.vacia():
          self.primero = self.primero.siguiente
          if self.primero:
              self.primero.anterior = None
          else:
              self.ultimo = None
          self.tamaño -= 1

  def eliminar_ultimo(self):
      if not self.vacia():
          self.ultimo = self.ultimo.anterior
          if self.ultimo:
              self.ultimo.siguiente = None
          else:
              self.primero = None
          self.tamaño -= 1

  def printear_inicio_lista(self):
    aux = self.primero
    while aux:
        print(aux.dato)
        aux = aux.siguiente

  def printear_final_lista(self):
    aux = self.ultimo
    while aux:
      print(aux.dato)
      aux = aux.anterior

  def vacia(self):
    return self.primero is None

lista = ListaDobleEnlazada()

lista.agregar_final(7)
lista.agregar_final(8)
lista.agregar_final(9)

print("Printear lista de inicio a fin, agregar final")
lista.printear_inicio_lista()
print("Printear lista de fin a inicio, agregar final")
lista.printear_final_lista()

lista.agregar_inicio(1)
lista.agregar_inicio(2)
lista.agregar_inicio(3)

print("Printear lista de inicio a fin, agregar inicio")
lista.printear_inicio_lista()
print("Printear lista de fin a inicio, agregar inicio")
lista.printear_final_lista()


lista.eliminar_primero()
print("Printear lista de inicio a fin, eliminar primero")
lista.printear_inicio_lista()
print("Printear lista de fin a inicio, eliminar primero")
lista.printear_final_lista()

lista.eliminar_ultimo()
print("Printear lista de inicio a fin, eliminar ultimo")
lista.printear_inicio_lista()

lista.printear_final_lista()
